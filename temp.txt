function M.Checkout(self,tab1,tab2)               --检测字段及其类型是否存在和是否正确
  for index,typename in pairs(tab2) do
    local temp_value = tab1.index
    if temp_value then                         --首先判断表1是否有此字段
      if typename == "int" then                --假如是数值类型
        local num = tonumber(temp_value)       --判断能不能转为数值类型
        if not num then                        --不能直返回
          return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
        end
      elseif typename == "string" then         --判断是不是字符串类型
        if not temp_value or type(temp_value) ~= "string" then
          return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
        end
      elseif typename == "map" then            --假如是个对象表类型
        local temp_fun = load('return' .. temp_value )
        local temp_tab = temp_fun()            --调用load中的字符串使得字符串生成表
        if temp_tab and type(temp_tab) == "table" then
          if #temp_tab == 0 then
            local ifture , message = self:Checkout(temp_value,typename)
            if not ifture then
              return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
            end
          else
            return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
          end
        else
          return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
        end
      elseif typename == "array" then          --假如是个数组表类型
        local temp_fun = load('return' .. temp_value )
        local temp_tab = temp_fun()            --调用load中的字符串使得字符串生成表
        if temp_tab and type(temp_tab) == "table" then
          local num = 0
          for i,v in pairs(temp_tab) do
            num = num + 1
          end
          if num == #temp_tab then
          else
            return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
          end 
        else
          return false , string.format("The Table's Type is not right : The %s type not %s", index,typename)
        end
      end
    else
      local return_str = string.format("The Table's Index is not exist : %s", index)
      return false , return_str                --假如没有字段返回false 和 缺失的字段
    end
  end
  return true
end
